# depth_map_by_DepthAnythingv2
Пример запуска: 
python main.py DJI_0193.MP4 \
  --subs flight.srt \
  --t_start 9 --t_end 60 \
  --gamma_fit \
  --roi_radius 40 \
  --cmap plasma_r \
  --annotate \
  --out processed_5_10s.mp4
 
  
  DJI_0193.MP4
— входное видео.

--subs flight.srt
— файл субтитров с телеметрией, содержащий строки вроде
H 12.34m (высота AGL).
Эти данные используются для покадровой калибровки карты расстояний.

--t_start 9 --t_end 60
— обрабатывать только часть видео с 9 по 60 секунду.
Скрипт пропускает всё до 9-й секунды и останавливается после 60-й.

--gamma_fit
— активирует автоматическую подстройку нелинейности γ для выравнивания карты глубины.
Это помогает уменьшить «перепады» на ровной поверхности (почва, поле и т.п.).

--roi_radius 40
— радиус области вокруг центра кадра (в пикселях),
используемой для калибровки и вычислений (земля под дроном).

--cmap plasma_r
— цветовая палитра matplotlib для визуализации глубины.
Здесь используется обратная “plasma”:
светлые области — ближе к камере, тёмные — дальше.
Можно выбрать и другие (gray_r, inferno_r, turbo_r, и др.).

--annotate
— добавляет в кадр текстовую аннотацию:
t=9.04s  H=12.35m  Dc≈12.10m
(время, высота над точкой взлёта, рассчитанное расстояние до земли).

--out processed_5_10s.mp4
— путь к выходному файлу с визуализацией результата.
